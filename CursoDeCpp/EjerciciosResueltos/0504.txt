Vamos a ir leyendo línea por línea:

char a = 'd';		---> a = 'd'
a -= 3;			---> a = a-3 --> a = 'd'-3 --> a = 'a' (porque a es char)

int b = 2;		---> b = 2

int c = b++ * 7;	---> c = b*7; b = b+1; --> c = 14, b = 3

int d = --b - ++c	---> b = b-1; c = c+1; d = b-c; --> b = 2, c = 15, d = -13

int e = b+1;		---> e = 2+1 --> e = 3


Resumiendo, al final de este fragmento de programa tenemos:
	a = 'a'
	b = 2
	c = 15
	d = -13
	e = 3
